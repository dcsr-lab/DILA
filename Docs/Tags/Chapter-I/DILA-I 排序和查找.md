    public int removeElement(int[] nums, int val) {
        if(nums.length == 0) return 0;

        int n = nums.length;
        Arrays.sort(nums); // æ’åº

        int start = 0, end = 0;
        int l = 0, r = n - 1;

            int mid = l + r >> 1;
            if(nums[mid] >= val) r = mid;
            else l = mid + 1;
        }

        if(nums[l] != val) return n;
        start = l;
        r = n - 1;

        while(l < r) { // å³è¾¹ç•Œ
            int mid = l + r  + 1>> 1;
            if(nums[mid] <= val) l = mid;
            else r = mid - 1;
        }
        end = r;

            int temp = nums[i];
            nums[i] = nums[j];
            nums[j] = temp;
