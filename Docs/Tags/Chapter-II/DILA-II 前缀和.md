    public int search(int[] s, int l, int r, double x) { // äºŒåˆ†æœç´¢å·¦è¾¹ç•Œ

    public double[] sampleStats(int[] count) {
        int n = count.length;

        int minimum = Integer.MAX_VALUE, maximum = Integer.MIN_VALUE; // æœ€å°å€¼å’Œæœ€å¤§å€¼

            if(count[i] != 0) {

        return a % b;
    }

    public boolean checkSubarraySum(int[] nums, int k) {
        if(nums.length < 2) return false;
        int n = nums.length;
        HashMap<Integer, Integer> hash = new HashMap<>();
        int[] s = new int[n + 10];

        hash.put(getMod(s[0], k), 0);
        for(int i = 1; i <= n; i ++){
            s[i] = s[i - 1] + nums[i - 1];

            int mod = getMod(s[i], k);
            if(hash.containsKey(mod)){

            } else hash.put(mod, i);
        }
        return false;
    }
}
